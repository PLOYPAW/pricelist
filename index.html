        // ใส่ URL ที่เป็น CSV format จาก Google Sheets

        const sheetUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vTgOz7WIAMIIwhmJ8jPy7ouaRtOkJddwn7EvZW_9z7CGdOJIsVW-FuXdBaZbWlwNXUJSR-SrAf9iBpM/pub?output=csv'; 


        let allProducts = [];

        let currentCategory = 'All';


        async function fetchData() {

            try {

                if (!sheetUrl) {

                    console.warn("Please provide a valid sheetUrl");

                    return;

                }

                const response = await fetch(sheetUrl);

                const data = await response.text();

                allProducts = csvToJSON(data);

                renderUI();

            } catch (error) {

                console.error("Error fetching data:", error);

                document.getElementById('loading').innerHTML = `<p class="text-pink-600 font-medium text-xl">การโหลดข้อมูลขัดข้อง (ตรวจสอบว่าได้ Publish as CSV หรือยัง?)</p>`;

            }

        }


        function csvToJSON(csv) {

            const lines = csv.split("\n");

            const result = [];

            // ดึง Headers และตัดช่องว่าง/ตัวอักษรพิเศษที่อาจติดมา

            const headers = lines[0].split(",").map(h => h.trim().replace(/[^\w]/gi, ''));

            

            for (let i = 1; i < lines.length; i++) {

                if (!lines[i]) continue;

                const obj = {};

                // ใช้ regex เพื่อจัดการ comma ภายในเครื่องหมายคำพูด (ถ้ามี)

                const currentline = lines[i].split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/);


                for (let j = 0; j < headers.length; j++) {

                    let value = currentline[j]?.trim() || '';

                    // ลบเครื่องหมายคำพูดที่ล้อมรอบค่า (ถ้ามี)

                    value = value.replace(/^"(.*)"$/, '$1');

                    obj[headers[j]] = value;

                }

                result.push(obj);

            }

            return result;

        }


        function renderUI() {

            document.getElementById('loading').classList.add('hidden');

            const categories = ['All', ...new Set(allProducts.map(p => p.Category).filter(Boolean))];

            filterContainer.innerHTML = categories.map(cat => `

                <button onclick="filterCategory('${cat}')" 

                    class="category-btn whitespace-nowrap px-10 py-5 rounded-full transition-all font-medium text-lg border-2 ${currentCategory === cat ? 'bg-pink-brand text-white border-pink-brand shadow-lg scale-105' : 'bg-white text-pink-400 border-pink-50 hover:bg-pink-50'}">

                    ${cat === 'All' ? 'สินค้าทั้งหมด ✨' : cat}

                </button>

            `).join('');

            updateGrid();

        }
